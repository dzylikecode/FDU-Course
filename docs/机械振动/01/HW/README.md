$$
\begin{aligned}
x_h &= e^{-\xi \omega_n t} ( \alpha e^{i \omega_d t} + \beta e^{-i \omega_d t} ) \\
&= e^{-\xi \omega_n t} [ ( \alpha + \beta ) \cos \omega_d t + i ( \alpha - \beta ) \sin \omega_d t ]
\end{aligned}
$$

$$
\begin{cases}
\alpha + \beta = \overline {\alpha + \beta} \\
i(\alpha - \beta) = \overline {i(\alpha - \beta)}
\end{cases}
$$
